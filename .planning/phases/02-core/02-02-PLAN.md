    在 static/js/app.js 中扩展，新增批处理功能。现有代码全部保留，按以下顺序修改和追加。

    === 1. state 对象扩展 ===

    === 3. 新增函数（按顺序追加在文件末尾） ===

    sleep(ms) -> Promise
    - return new Promise(resolve => setTimeout(resolve, ms))

    loadCheckpoint() -> async
    - fetch('/api/checkpoint') -> await res.json() -> state.processedBvids = new Set(data.processed || [])
    - catch: state.processedBvids = new Set()（静默失败）
    - 在 loadFile() 中: state.videos = data.videos 之后、renderVideoList(data.videos) 之前插入 await loadCheckpoint()

    updateButtons()
    - state.running 为 true 时: startBtn.disabled = true; stopBtn.disabled = false
    - false 时: startBtn.disabled = false; stopBtn.disabled = true

    updateProgress()
    - 总处理数 = state.stats.success + state.stats.failed + state.stats.skipped
    - 百分比 = 总处理数 / state.videos.length * 100（videos.length 为 0 时取 0）
    - progressBar.style.width = 百分比 + '%'
    - totalSpan.textContent = state.videos.length
    - successSpan.textContent = state.stats.success
    - failedSpan.textContent = state.stats.failed
    - skippedSpan.textContent = state.stats.skipped
    - currentSpan.textContent = state.running ? (state.currentIndex + 1) : '---'

    addLog(message, level)
    - level 取值: 'info' / 'success' / 'error' / 'skip' / 'warning'
    - 创建 div.log-entry, 添加类 log-{level}
    - 文本内容: 当前时间 [HH:MM:SS] 前缀 + message（使用 new Date().toLocaleTimeString('zh-CN', {hour12:false})）
    - appendChild 到 logArea, logArea.scrollTop = logArea.scrollHeight
    - 所有文本内容使用 textContent 而非 innerHTML（防 XSS, T-02-UI-02）

    submitVideo(video) -> async Promise(taskId)
    - 从 DOM 读取当前配置: providerSelect.value, modelSelect.value, styleSelect.value, promptInput.value
    - fetch POST /api/process-video, headers Content-Type application/json, body JSON.stringify({ bvid, title, folder, url, model_name, provider_id, style, extras: promptInput.value })
    - const data = await res.json()
    - if (!res.ok) throw new Error(data.error || '提交失败')
    - return data.task_id

    pollTaskStatus(taskId, timeoutMs=900000) -> async Promise(statusData)
    - 使用 new Promise(resolve, reject) + setInterval 实现
    - const startTime = Date.now()
    - const interval = setInterval(async () => {...}, 3000) 每 3 秒轮询（D-02）
    - 每次轮询:
      - 检查 Date.now() - startTime > timeoutMs: clearInterval, reject(new Error('任务超时'))
      - try: fetch('/api/task-status/' + taskId), await res.json()
        - if data.status === 'SUCCESS' || data.status === 'FAILED': clearInterval, resolve(data)
        - else: 继续轮询
      - catch(err): clearInterval, reject(err)
    - 三条路径（resolve/reject/timeout）都必须调用 clearInterval（防 zombie 轮询, RESEARCH Pitfall 4）

    stopBatch()
    - state.stopRequested = true
    - addLog('用户请求停止，完成当前任务后结束...', 'warning')
    - stopBtn.disabled = true

    startBatch() -> async
    - 守卫: if (state.running) return（防并行, Pitfall 3）
    - state.running = true; state.stopRequested = false
    - state.currentIndex = 0
    - const videos = state.videos（局部引用）
    - state.stats = { total: videos.length, success: 0, failed: 0, skipped: 0 }
    - progressSection.style.display = 'block'; logArea.innerHTML = ''
    - updateButtons(); addLog('批量处理开始', 'info')
    - state.outputDir = outputDirInput.value.trim() || './output'
    - for (let i = 0; i < videos.length; i++):
      - state.currentIndex = i
      - if (!state.running || state.stopRequested) break
      - updateProgress()
      - const video = videos[i]
      - 检查 state.processedBvids.has(video.bvid):
        - 是: addLog('[i+1/videos.length] 跳过已处理: ' + escapeHtml(video.title), 'skip'); state.stats.skipped++; continue
      - addLog('[i+1/videos.length] ' + escapeHtml(video.title || video.bvid), 'info')
      - try:
        - taskId = await submitVideo(video)
        - result = await pollTaskStatus(taskId, 900000)
        - if result.status === 'SUCCESS':
          - addLog('完成: ' + escapeHtml(video.title || video.bvid), 'success')
          - state.processedBvids.add(video.bvid)
          - state.stats.success++
        - else:
          - addLog('失败: ' + escapeHtml(video.title || video.bvid), 'error')
          - state.stats.failed++
      - catch(err):
        - addLog('错误: ' + escapeHtml(video.title || video.bvid) + ' - ' + err.message, 'error')
        - state.stats.failed++
      - updateProgress()
      - if (i < videos.length - 1 && state.running && !state.stopRequested):
        - await sleep(3000)（CORE-07 冷却）
    - state.running = false; updateButtons(); addLog('批量处理结束', 'info')

    === 4. renderVideoList 状态标签更新 ===
    修改 renderVideoList 中状态 badge 的构建逻辑:
    - 如果 state.processedBvids.has(v.bvid): 使用 span.status-badge.status-success（文字"已处理"）
    - 否则使用 span.status-badge.status-pending（文字"待处理"）
    - 新增 CSS: .status-success { color: var(--success); background: rgba(64,192,87,0.15); }
